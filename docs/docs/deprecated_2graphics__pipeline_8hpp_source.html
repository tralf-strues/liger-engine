170/* Color attachment blending */
171/**
172 * @brief Specifies blending factor.
173 *
174 * Let
175 * 1. R_src, G_src, B_src, A_src - source color components
176 * 2. R_dst, G_dst, B_dst, A_dst - destination color components
177 *
178 * Then factors are defined as follows:
179 *
180 * Factor | RGB blend factors | Alpha blend factor |
181 * -----------------:|:---------------------------------:|:------------------------------:|
182 * kZero | (0, 0, 0) | 0 |
183 * kOne | (1, 1, 1) | 1 |
184 * kSrcColor | (R_src, G_src, B_src) | A_src |
185 * kOneMinusSrcColor | (1 - R_src, 1 - G_src, 1 - B_src) | 1 - A_src |
186 * kDstColor | (R_dst, G_dst, B_dst) | A_dst |
187 * kOneMinusDstColor | (1 - R_dst, 1 - G_dst, 1 - B_dst) | 1 - A_dst |
188 * kSrcAlpha | (A_src, A_src, A_src) | A_src |
189 * kOneMinusSrcAlpha | (1 - A_src, 1 - A_src, 1 - A_src) | 1 - A_src |
190 * kDstAlpha | (A_dst, A_dst, A_dst) | A_dst |
191 * kOneMinusDstAlpha | (1 - A_dst, 1 - A_dst, 1 - A_dst) | 1 - A_dst |
192 */
193DECLARE_ENUM_CLASS(ColorBlendFactor, uint32,
194 kZero, kOne,
195
196 kSrcColor, kOneMinusSrcColor, kDstColor, kOneMinusDstColor,
197
198 kSrcAlpha, kOneMinusSrcAlpha, kDstAlpha, kOneMinusDstAlpha);
199
200/**
201 * @brief Specifies blending operation.
202 *
203 * Let
204 * 1. R_src, G_src, B_src, A_src - source color components
205 * 2. R_dst, G_dst, B_dst, A_dst - destination color components
206 * 3. SF_r, SF_g, SF_b, SF_a - source blend factor components
207 * 4. DF_r, DF_g, DF_b, DF_a - destination blend factor components
208 *
209 * Then operations are defined as follows:
210 *
211 * Operation | Final R/G/B | Final A |
212 * ----------------:|:------------------------------:|:------------------------------:|
213 * kAdd | R_src * SF_r + R_dst * DF_r | A_src * SF_a + A_dst * DF_a |
214 * kSubtract | R_src * SF_r - R_dst * DF_r | A_src * SF_a - A_dst * DF_a |
215 * kReverseSubtract | R_dst * DF_r - R_src * SF_r | A_dst * DF_a - A_src * SF_a |
216 * kMin | min(R_src, R_dst) | min(A_src, A_dst) |
217 * kMax | max(R_src, R_dst) | max(A_src, A_dst) |
218 */
219DECLARE_ENUM_CLASS(ColorBlendOperation, uint32, kAdd, kSubtract, kReverseSubtract, kMin, kMax);
