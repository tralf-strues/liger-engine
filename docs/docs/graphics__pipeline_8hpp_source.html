123struct ColorBlendInfo {
124 /**
125 * @brief Specifies blending factor.
126 *
127 * Let
128 * 1. R_src, G_src, B_src, A_src - source color components
129 * 2. R_dst, G_dst, B_dst, A_dst - destination color components
130 *
131 * Then factors are defined as follows:
132 *
133 * Factor | RGB blend factors | Alpha blend factor |
134 * -----------------:|:---------------------------------:|:------------------------------:|
135 * kZero | (0, 0, 0) | 0 |
136 * kOne | (1, 1, 1) | 1 |
137 * -----------------:|:---------------------------------:|:------------------------------:|
138 * kSrcColor | (R_src, G_src, B_src) | A_src |
139 * kOneMinusSrcColor | (1 - R_src, 1 - G_src, 1 - B_src) | 1 - A_src |
140 * kDstColor | (R_dst, G_dst, B_dst) | A_dst |
141 * kOneMinusDstColor | (1 - R_dst, 1 - G_dst, 1 - B_dst) | 1 - A_dst |
142 * -----------------:|:---------------------------------:|:------------------------------:|
143 * kSrcAlpha | (A_src, A_src, A_src) | A_src |
144 * kOneMinusSrcAlpha | (1 - A_src, 1 - A_src, 1 - A_src) | 1 - A_src |
145 * kDstAlpha | (A_dst, A_dst, A_dst) | A_dst |
146 * kOneMinusDstAlpha | (1 - A_dst, 1 - A_dst, 1 - A_dst) | 1 - A_dst |
147 */
148 enum class Factor : uint8_t {
149 kZero,
150 kOne,
151
152 kSrcColor,
153 kOneMinusSrcColor,
154 kDstColor,
155 kOneMinusDstColor,
156
157 kSrcAlpha,
158 kOneMinusSrcAlpha,
159 kDstAlpha,
160 kOneMinusDstAlpha
161 };

163 /**
164 * @brief Specifies blending operation.
165 *
166 * Let
167 * 1. R_src, G_src, B_src, A_src - source color components
168 * 2. R_dst, G_dst, B_dst, A_dst - destination color components
169 * 3. SF_r, SF_g, SF_b, SF_a - source blend factor components
170 * 4. DF_r, DF_g, DF_b, DF_a - destination blend factor components
171 *
172 * Then operations are defined as follows:
173 *
174 * Operation | Final R/G/B | Final A |
175 * ----------------:|:------------------------------:|:------------------------------:|
176 * kAdd | R_src * SF_r + R_dst * DF_r | A_src * SF_a + A_dst * DF_a |
177 * kSubtract | R_src * SF_r - R_dst * DF_r | A_src * SF_a - A_dst * DF_a |
178 * kReverseSubtract | R_dst * DF_r - R_src * SF_r | A_dst * DF_a - A_src * SF_a |
179 * kMin | min(R_src, R_dst) | min(A_src, A_dst) |
180 * kMax | max(R_src, R_dst) | max(A_src, A_dst) |
181 */
182 enum class Operation : uint8_t {
183 kAdd,
184 kSubtract,
185 kReverseSubtract,
186 kMin,
187 kMax
188 };
