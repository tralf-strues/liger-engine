65template <typename EventT>
66EventSink<EventT>& EventDispatcher::GetSink() {
67 EventSink<EventT>* sink = nullptr;
68
69 detail::EventTypeId type_id = detail::EventTypeIdHolder<EventT>::Value();
70
71 auto it = sinks_.find(type_id);
72 if (it == sinks_.end()) {
73 it = sinks_.emplace(type_id, std::move(std::make_unique<EventSink<EventT>>())).first;
74 }
75
76 return *dynamic_cast<EventSink<EventT>*>(it->second.get());
77}

79template <typename EventT>
80bool EventDispatcher::Dispatch(const EventT& event, bool dispatch_to_all) {
81 return GetSink<EventT>().Dispatch(event, dispatch_to_all);
82}
