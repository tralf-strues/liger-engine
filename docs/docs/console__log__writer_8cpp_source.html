39void ConsoleLogWriter::OnMessageAdded(const LogMessage& message) {
40 if (style_.write_level) {
41 fmt::print(GetLevelStyle(message.level), "{0} ", GetLevelName(message.level));
42 }
43
44 if (style_.write_channel) {
45 fmt::print(GetTextStyle(message.level), "{0} ", GetChannelName(message.channel));
46 }
47
48 fmt::print(GetTextStyle(message.level), "{0}\n", message.message);
49}

55const fmt::text_style& ConsoleLogWriter::GetLevelStyle(LogLevel level) const {
