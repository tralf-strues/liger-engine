35Window::Window(uint32_t width, uint32_t height, const std::string_view title) {
36 glfwWindowHint(GLFW_CLIENT_API, GLFW_NO_API);
37 glfwWindowHint(GLFW_RESIZABLE, GLFW_TRUE);
38
39 window_ = glfwCreateWindow(width, height, title.data(), nullptr, nullptr);
40 LIGER_ASSERT(window_, "Window", "glfwCreateWindow failed!");
41}
42
43Window::~Window() {
44 glfwDestroyWindow(window_);
45}
46
47GLFWwindow* Window::GetGLFWwindow() { return window_; }

49void Window::SetTitle(const std::string_view title) {

51}

53uint32_t Window::GetWidth() const {
