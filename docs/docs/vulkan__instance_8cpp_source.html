42bool CheckValidationLayerSupport() {
43 uint32_t layer_count = 0;
44 VULKAN_CALL(vkEnumerateInstanceLayerProperties(&layer_count, nullptr));
45
46 std::vector<VkLayerProperties> available_layers(layer_count);
47 VULKAN_CALL(vkEnumerateInstanceLayerProperties(&layer_count, available_layers.data()));
48
49 bool layer_found = false;
50
51 for (const auto& layer_properties : available_layers) {
52 if (strcmp(layer_properties.layerName, kValidationLayerName) == 0) {
53 layer_found = true;
54 break;
55 }
56 }
57
58 return layer_found;
59}

61std::vector<const char*> GetInstanceExtensions() {
62 std::vector<const char*> extensions;
63
64 /* GLFW extensions */
65 uint32_t glfw_extension_count = 0;
66 const char** glfw_extensions = glfwGetRequiredInstanceExtensions(&glfw_extension_count);
67
68 for (uint32_t i = 0; i < glfw_extension_count; ++i) {
69 extensions.push_back(glfw_extensions[i]);
70 }
71
72#ifdef __APPLE__
73 extensions.push_back(VK_MVK_MACOS_SURFACE_EXTENSION_NAME);
74 extensions.push_back(VK_KHR_PORTABILITY_ENUMERATION_EXTENSION_NAME);
75 extensions.push_back(VK_KHR_GET_PHYSICAL_DEVICE_PROPERTIES_2_EXTENSION_NAME);
76#endif
77
78 extensions.push_back(VK_EXT_VALIDATION_FEATURES_EXTENSION_NAME);
79
80 return extensions;
81}

87bool VulkanInstance::Init(ValidationLevel validation) {
88 if (validation != IInstance::ValidationLevel::kNone && !CheckValidationLayerSupport()) {
89 validation = IInstance::ValidationLevel::kNone;
90 LIGER_LOG_ERROR(kLogChannelRHI, "Validation layer \"{0}\" is not found", kValidationLayerName);
91 }
92
93 const VkApplicationInfo app_info{
94 .sType = VK_STRUCTURE_TYPE_APPLICATION_INFO,
95 .pNext = nullptr,
96 .pApplicationName = nullptr,
97 .applicationVersion = 0,
98 .pEngineName = "Liger Engine",
99 .engineVersion = VK_MAKE_VERSION(1, 0, 0),
100 .apiVersion = VK_API_VERSION_1_3
101 };
102
103 auto extensions = GetInstanceExtensions();
104
105 VkInstanceCreateInfo instance_info{
106 .sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO,
107 .pNext = nullptr,
108 .flags = 0,
109 .pApplicationInfo = &app_info,
110 .enabledLayerCount = 0,
111 .ppEnabledLayerNames = nullptr,
112 .enabledExtensionCount = static_cast<uint32_t>(extensions.size()),
113 .ppEnabledExtensionNames = extensions.data()
114 };
115
116#ifdef __APPLE__
117 instance_info.flags = VK_INSTANCE_CREATE_ENUMERATE_PORTABILITY_BIT_KHR;
118#endif
119
120 if (validation != IInstance::ValidationLevel::kNone) {
121 instance_info.enabledLayerCount = 1;
122 instance_info.ppEnabledLayerNames = &kValidationLayerName;
123 }
124
125 const VkValidationFeatureEnableEXT extra_features[] = {
126 VK_VALIDATION_FEATURE_ENABLE_GPU_ASSISTED_EXT,
127 VK_VALIDATION_FEATURE_ENABLE_BEST_PRACTICES_EXT,
128 VK_VALIDATION_FEATURE_ENABLE_SYNCHRONIZATION_VALIDATION_EXT
129 };
130
131 VkValidationFeaturesEXT features_info{
132 .sType = VK_STRUCTURE_TYPE_VALIDATION_FEATURES_EXT,
133 .pNext = nullptr,
134 .enabledValidationFeatureCount = std::size(extra_features),
135 .pEnabledValidationFeatures = extra_features,
136 .disabledValidationFeatureCount = 0,
137 .pDisabledValidationFeatures = nullptr
138 };
139
140 if (validation == IInstance::ValidationLevel::kExtensive) {
141 instance_info.pNext = &features_info;
142 }
143
144 VULKAN_CALL(vkCreateInstance(&instance_info, nullptr, &instance_));
145
146 return true;
147}

157bool VulkanInstance::FillDeviceInfoList() {
158 uint32_t device_count = 0;
159 VULKAN_CALL(vkEnumeratePhysicalDevices(instance_, &device_count, nullptr));
160
161 if (device_count == 0) {
162 LIGER_LOG_ERROR(kLogChannelRHI, "There are no Vulkan physical devices found!");
163 return false;
164 }
165
166 std::vector<VkPhysicalDevice> physical_devices(device_count);
167 VULKAN_CALL(vkEnumeratePhysicalDevices(instance_, &device_count, physical_devices.data()));
168
169 for (auto physical_device : physical_devices) {
170 VkPhysicalDeviceFeatures features;
171 vkGetPhysicalDeviceFeatures(physical_device, &features);
172
173 VkPhysicalDeviceProperties properties;
174 vkGetPhysicalDeviceProperties(physical_device, &properties);
175
176 uint32_t extensions_count = 0;
177 VULKAN_CALL(vkEnumerateDeviceExtensionProperties(physical_device, nullptr, &extensions_count, nullptr));
178
179 std::vector<VkExtensionProperties> available_extensions{extensions_count};
180 VULKAN_CALL(vkEnumerateDeviceExtensionProperties(physical_device, nullptr, &extensions_count,
181 available_extensions.data()));
182
183 bool required_extensions_supported = true;
184 for (auto required_extension : kRequiredDeviceExtensions) {
185 bool found = false;
186
187 for (auto available_extension : available_extensions) {
188 if (std::strcmp(required_extension, available_extension.extensionName) == 0) {
189 found = true;
190 break;
191 }
192 }
193
194 if (!found) {
195 required_extensions_supported = false;
196 break;
197 }
198 }
199
200 bool swapchain_supported = false;
201 for (auto available_extension : available_extensions) {
202 if (std::strcmp(available_extension.extensionName, VK_KHR_SWAPCHAIN_EXTENSION_NAME) == 0) {
203 swapchain_supported = true;
204 break;
205 }
206 }
207
208 IDevice::Info info{
209 .id = properties.deviceID,
210 .name = properties.deviceName,
211 .type = GetDeviceTypeFromVulkan(properties.deviceType),
212 .engine_supported = required_extensions_supported,
213
214 .properties = {
215 .swapchain = swapchain_supported,
216 .sampler_anisotropy = static_cast<bool>(features.samplerAnisotropy),
217 .max_msaa_samples = GetMaxSamplesFromVulkan(properties),
218 .max_sampler_anisotropy = properties.limits.maxSamplerAnisotropy
219 }
220 };
221
222 device_info_list_.emplace_back(std::move(info));
223 }
224
225 return true;
226}
