43/**
44 * @brief Event sink for callbacks of the specified event.
45 *
46 * @tparam EventT Event type.
47 */
48template <typename EventT>
49class EventSink final : public detail::IBaseEventSink {
50 private:
51 /**
52 * @brief Callback type, returns whether or not the event has been handled.
53 */
54 using CallbackFunctionT = bool(const EventT&);
55
56 public:
57 ~EventSink() override = default;
58
59 /**
60 * @brief Add function callback.
61 *
62 * @tparam FunctionT
63 */
64 template <auto FunctionT>
65 void Connect() {
66 detail::Callback<CallbackFunctionT> callback;
67 callback.template Connect<FunctionT>();
68 callbacks_.emplace_back(std::move(callback));
69 }
70
71 /**
72 * @brief Add class method callback.
73 *
74 * @tparam FunctionT
75 * @tparam ClassT
76 * @param instance
77 */
78 template <auto FunctionT, class ClassT>
79 void Connect(ClassT& instance) {
80 detail::Callback<CallbackFunctionT> callback;
81 callback.template Connect<FunctionT, ClassT>(instance);
82 callbacks_.emplace_back(std::move(callback));
83 }
84
85 /**
86 * @brief Remove function callback.
87 *
88 * @tparam FunctionT
89 */
90 template <auto FunctionT>
91 void Remove() {
92 detail::Callback<CallbackFunctionT> callback;
93 callback.template Connect<FunctionT>();
94
95 for (auto it = callbacks_.begin(); it != callbacks_.end(); ++it) {
96 if (*it == callback) {
97 callbacks_.erase(it);
98 break;
99 }
100 }
101 }
102
103 /**
104 * @brief Remove class method callback.
105 *
106 * @tparam FunctionT
107 * @tparam ClassT
108 * @param instance
109 */
110 template <auto FunctionT, class ClassT>
111 void Remove(ClassT& instance) {
112 detail::Callback<CallbackFunctionT> callback;
113 callback.template Connect<FunctionT, ClassT>(instance);
114
115 for (auto it = callbacks_.begin(); it != callbacks_.end(); ++it) {
116 if (*it == callback) {
117 callbacks_.erase(it);
118 break;
119 }
120 }
121 }
122
123 /**
124 * @brief Dispatch the event to all callbacks.
125 *
126 * @param event
127 * @param dispatch_to_all
128 *
129 * @return Whether or not the event has been handled.
130 */
131 bool Dispatch(const EventT& event, bool dispatch_to_all = false) {
132 bool event_handled = false;
133
134 for (size_t i = 0; i < callbacks_.size(); ++i) {
135 event_handled = event_handled || callbacks_[i](event);
136
137 if (event_handled && !dispatch_to_all) {
138 break;
139 }
140 }
141
142 return event_handled;
143 }
144
145 private:
146 std::vector<detail::Callback<CallbackFunctionT>> callbacks_;
147};
