134template <typename Node>
135bool DAG<Node>::TopologicalSort() {
136 std::vector<bool> visited(nodes_.size(), false);
137 std::vector<bool> on_stack(nodes_.size(), false);
138
139 sorted_.clear();
140
141 for (NodeHandle from_handle = 0; from_handle < static_cast<NodeHandle>(nodes_.size()); ++from_handle) {
142 if (visited[from_handle]) {
143 continue;
144 }
145
146 if (!TopologicalSortDFS(from_handle, visited, on_stack)) {
147 return false;
148 }
149 }
150
151 std::reverse(sorted_.begin(), sorted_.end());
152 return true;
153}

155template <typename Node>
156bool DAG<Node>::TopologicalSortDFS(NodeHandle from_handle, DFSVisitedList& visited, DFSOnStackList& on_stack) {
157 visited[from_handle] = true;
158 on_stack[from_handle] = true;
159
160 for (NodeHandle to_handle : adj_lists_[from_handle]) {
161 if (visited[to_handle] && on_stack[to_handle]) {
162 return false;
163 }
164
165 if (visited[to_handle]) {
166 continue;
167 }
168
169 if (!TopologicalSortDFS(to_handle, visited, on_stack)) {
170 return false;
171 }
172 }
173
174 on_stack[from_handle] = false;
175 sorted_.push_back(from_handle);
176
177 return true;
178}

180template <typename Node>
181void DAG<Node>::UpdateDepths() {
182 depths_.resize(nodes_.size());
183 std::fill(depths_.begin(), depths_.end(), 0);
184
185 for (auto from_handle : sorted_) {
186 for (auto to_handle : adj_lists_[from_handle]) {
187 depths_[to_handle] = std::max(depths_[to_handle], depths_[from_handle] + 1);
188 }
189 }
190}
