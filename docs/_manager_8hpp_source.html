69 
70  std::unique_lock<std::mutex> lock(mutex_);
71 
72  auto handle = storage_.Get<Asset>(id);
73  if (handle) {
74  return handle;
75  }
76 
77  auto filepath = registry_.GetAbsoluteFile(id);

80  LIGER_ASSERT(loader, kLogChannelAsset, "No loader for extension '{0}' found", extension.string());
