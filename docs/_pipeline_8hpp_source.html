124 struct ColorBlendInfo {
125  /**
126  * @brief Specifies blending factor.
127  *
128  * Let
129  * 1. R_src, G_src, B_src, A_src - source color components
130  * 2. R_dst, G_dst, B_dst, A_dst - destination color components
131  *
132  * Then factors are defined as follows:
133  *
134  * Factor | RGB blend factors | Alpha blend factor |
135  * -----------------:|:---------------------------------:|:------------------------------:|
136  * kZero | (0, 0, 0) | 0 |
137  * kOne | (1, 1, 1) | 1 |
138  * -----------------:|:---------------------------------:|:------------------------------:|
139  * kSrcColor | (R_src, G_src, B_src) | A_src |
140  * kOneMinusSrcColor | (1 - R_src, 1 - G_src, 1 - B_src) | 1 - A_src |
141  * kDstColor | (R_dst, G_dst, B_dst) | A_dst |
142  * kOneMinusDstColor | (1 - R_dst, 1 - G_dst, 1 - B_dst) | 1 - A_dst |
143  * -----------------:|:---------------------------------:|:------------------------------:|
144  * kSrcAlpha | (A_src, A_src, A_src) | A_src |
145  * kOneMinusSrcAlpha | (1 - A_src, 1 - A_src, 1 - A_src) | 1 - A_src |
146  * kDstAlpha | (A_dst, A_dst, A_dst) | A_dst |
147  * kOneMinusDstAlpha | (1 - A_dst, 1 - A_dst, 1 - A_dst) | 1 - A_dst |
148  */
149  enum class Factor : uint8_t {
150  Zero,
151  One,
152 
153  SrcColor,
154  OneMinusSrcColor,
155  DstColor,
156  OneMinusDstColor,
157 
158  SrcAlpha,
159  OneMinusSrcAlpha,
160  DstAlpha,
161  OneMinusDstAlpha
162  };
163 
164  /**
165  * @brief Specifies blending operation.
166  *
167  * Let
168  * 1. R_src, G_src, B_src, A_src - source color components
169  * 2. R_dst, G_dst, B_dst, A_dst - destination color components
170  * 3. SF_r, SF_g, SF_b, SF_a - source blend factor components
171  * 4. DF_r, DF_g, DF_b, DF_a - destination blend factor components
172  *
173  * Then operations are defined as follows:
174  *
175  * Operation | Final R/G/B | Final A |
176  * ----------------:|:------------------------------:|:------------------------------:|
177  * kAdd | R_src * SF_r + R_dst * DF_r | A_src * SF_a + A_dst * DF_a |
178  * kSubtract | R_src * SF_r - R_dst * DF_r | A_src * SF_a - A_dst * DF_a |
179  * kReverseSubtract | R_dst * DF_r - R_src * SF_r | A_dst * DF_a - A_src * SF_a |
180  * kMin | min(R_src, R_dst) | min(A_src, A_dst) |
181  * kMax | max(R_src, R_dst) | max(A_src, A_dst) |
182  */
183  enum class Operation : uint8_t {
184  Add,
185  Subtract,
186  ReverseSubtract,
187  Min,
188  Max
189  };
