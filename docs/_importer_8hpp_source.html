35 /**
36  * @brief Importer interface.
37  *
38  * Importers are basically converters of one asset file (based on extension) to possibly several
39  * asset files better suited for some particular usage. An example can be an FBX importer that
40  * generates mesh/material/texture/etc. files in a format which is tailored specifically for a
41  * particular renderer implementation.
42  */
43 class IImporter {
44  public:
45  /**
46  * @brief Import result which (if successful) lists imported files and dependencies between them.
47  */
48  struct Result {
49  /**
50  * @brief Whether import has finished successfully.
51  */
52  bool success{false};
53 
54  /**
55  * @brief List of filepaths of imported assets.
56  *
57  * @note The filepaths are based on the import destination folder. For instance, if `dst_folder`
58  * is "assets/imported/", and the importer generated file "teapot.mesh", then the filepath
59  * will be "assets/imported/teapot.mesh".
60  */
61  std::vector<std::filesystem::path> files;
62 
63  /**
64  * @brief List of dependencies between assets.
65  *
66  * Each dependency is represented by two indices into @ref files list. The first index is the
67  * index of the dependent file and the second one is of its dependency file.
68  */
69  std::vector<std::pair<uint32_t, uint32_t>> dependencies;
70  };
71 
72  virtual ~IImporter() = default;
73 
74  /**
75  * @brief File extension of files which this importer can import, e.g. ('.fbx').
76  */
77  virtual const std::filesystem::path& FileExtension() const = 0;
78 
79  /**
80  * @brief Try to import `src` file and save generated files to `dst_folder`.
81  */
82  virtual Result Import(const std::filesystem::path& src, const std::filesystem::path& dst_folder) const = 0;
83 };
