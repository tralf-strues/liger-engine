41 class EventSink final : public detail::IBaseEventSink {
42  private:
43  /**
44  * @brief Callback type, returns whether or not the event has been handled.
45  */
46  using CallbackFunctionT = bool(const EventT&);
47 
48  public:

51  /**
52  * @brief Add function callback.
53  *
54  * @tparam FunctionT
55  */
56  template <auto FunctionT>
57  void Connect() {
58  detail::Callback<CallbackFunctionT> callback;
59  callback.template Connect<FunctionT>();
60  callbacks_.emplace_back(std::move(callback));
61  }

63  /**

77  /**
78  * @brief Remove function callback.
79  *
80  * @tparam FunctionT
81  */
82  template <auto FunctionT>
83  void Remove() {
84  detail::Callback<CallbackFunctionT> callback;
85  callback.template Connect<FunctionT>();
86 
87  for (auto it = callbacks_.begin(); it != callbacks_.end(); ++it) {
88  if (*it == callback) {
89  callbacks_.erase(it);
90  break;
91  }
92  }
93  }

95  /**
96  * @brief Remove class method callback.
97  *
98  * @tparam FunctionT
99  * @tparam ClassT
100  * @param instance
101  */
102  template <auto FunctionT, class ClassT>
103  void Remove(ClassT& instance) {
104  detail::Callback<CallbackFunctionT> callback;
105  callback.template Connect<FunctionT, ClassT>(instance);
106 
107  for (auto it = callbacks_.begin(); it != callbacks_.end(); ++it) {
108  if (*it == callback) {
109  callbacks_.erase(it);
110  break;
111  }
112  }
113  }
114 
115  /**
116  * @brief Dispatch the event to all callbacks.
117  *
118  * @param event
119  * @param dispatch_to_all
120  *
121  * @return Whether or not the event has been handled.
122  */
123  bool Dispatch(const EventT& event, bool dispatch_to_all = false) {
124  bool event_handled = false;
125 
126  for (size_t i = 0; i < callbacks_.size(); ++i) {
127  event_handled = event_handled || callbacks_[i](event);
128 
129  if (event_handled && !dispatch_to_all) {
130  break;
131  }
132  }
133 
134  return event_handled;
135  }
136 
137  private:
138  std::vector<detail::Callback<CallbackFunctionT>> callbacks_;
139 };
